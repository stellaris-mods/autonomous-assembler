ass_stage2_system_reached = {
	# We dont need the event target, it's just much easier than prevprevprevprevprevprevprevprev
	PREVPREVPREV = { save_event_target_as = ass_target }
	if = {
		limit = { ass_has_operation = yes }
		if = {
			limit = {
				solar_system = {
					is_same_value = event_target:ass_target
				}
			}
			log = "[this.GetName]: In system [ass_target.GetName]."
			ass_init_stage3 = yes
			else = {
				log = "[this.GetName]: ERROR: We are not in system [ass_target.GetName]."
				event_target:ass_target = {
					set_timed_star_flag = {
						flag = "ass_error"
						days = 180
					}
				}
				ass_op_abort = yes
			}
		}
		else = {
			log = "[this.GetName]: ERROR: Reached end of queue_actions without an operation. That's weird."
			event_target:ass_target = {
				set_timed_star_flag = {
					flag = "ass_error"
					days = 180
				}
			}
			ass_op_abort = yes
		}
	}
}

ass_stage2_system_move = {
	PREV = {
		fleet = {
			set_fleet_flag = "autonomous_initial_order" # Removed instantly
			set_fleet_flag = "autonomous_operation"
			queue_actions = {
				move_to = PREVPREV
				repeat = {
					while = {
						id = assSystemMove.1
						NOT = { is_fleet_idle = yes }
					}
					wait = 7
				}
				effect = {
					id = assSystemMove.2
					PREV = { ass_stage2_system_reached = yes }
				}
			}
		}
	}
}
