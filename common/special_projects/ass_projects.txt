special_project = {
	event_chain = "autonomous_assembler_setup"
	timelimit = -1
	cost = 0
	days_to_research = 7
	tech_department = engineering_technology
	event_scope = planet_event
	on_fail = {}
	location = no
	picture = GFX_evt_space_station

	icon = "gfx/interface/icons/situation_log/situation_log_engineering.dds"
	key = "ass_go_home"
	on_success = {
		owner = {
			log = "ASS: Go home to planet and dock for 3 months."
			# Prevents any new operation from being undertaken
			# We also set a timed flag on each ship when it reaches home
			set_timed_country_flag = {
				flag = "ass_halt_operations"
				days = 90
			}
			# Prevents this project for showing up again for the next 90 days
			set_country_flag = "ass_go_home"

			# Make any owned ship head for home
			every_owned_ship = {
				limit = { is_ship_size = autonomous_assembler }
				if = {
					limit = { ass_has_mission = yes }
					ass_mission_abort = yes
				}
				ass_set_mission_type = yes
			}
			country_event = { id = assCountry.500 }

			# Just remove ass_go_home
			country_event = { id = assCountry.400 days = 90 }
		}
	}
}
