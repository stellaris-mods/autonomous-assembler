namespace = assSpaceportModules

# on_ship_built & on_ship_upgraded
# Root = Ship
ship_event = {
	id = assSpaceportModules.1
	title = "ass_spaceport_module.explain.title"
	desc = "ass_spaceport_module.explain.desc"
	picture = GFX_evt_autonomous_assembler
	is_triggered_only = yes
	trigger = {
		is_ship_size = autonomous_assembler
		has_component = ASS_EXPANSION_SPACEPORT_MODULES
		NOT = { owner = { any_owned_planet = { ass_is_blueprint_planet = yes } } }
	}
	option = { name = OK }
}

# Spaceport Module operation
ship_event = {
	id = assSpaceportModules.1000
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				exists = fleet
				fleet = { exists = solar_system }
			}
			if = {
				limit = { fleet = { check_variable = { which = "AssConstructionTime" value = 0 } } }

				# We are done with building, spawn it
				remove_ship_flag = "ass_constructing_station"
				ass_stage4_complete_spaceport_module = yes

				else = {
					# Nope, wait some more.
					fleet = {
						subtract_variable = { which = "AssConstructionTime" value = 10 }
					}
					ship_event = { id = assSpaceportModules.1000 days = 10 }
				}
			}
			else = {
				# No fleet/solar system, so abort!
				ass_stage4_abort_operation = yes
			}
		}
	}
}
